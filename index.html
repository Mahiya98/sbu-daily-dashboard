/*************** HTML BUILDER (Polished) ***************/ function buildHtml(data, analysis) { const today = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "dd-MMM-yyyy"); // ── Helpers ───────────────────────────────────────────────────────────── const fmtPct = v => (v === null || v === undefined || isNaN(v)) ? "—" : v + "%"; const ach = analysis.byAchievement; const oee = analysis.byOEE; // Tile builder for best/worst highlight cards function tile(label, sbu, kpiLabel, kpiValue, gradFrom, gradTo, accent) { return `
${label}
${sbu ? sbu.name : "—"}
${kpiLabel}: ${sbu ? fmtPct(sbu[kpiValue]) : "—"}
`; } // Ranking table builder (top 3 + bottom 1 style) function rankTable(title, sortedList, field, color) { if (!sortedList.length) return ""; const rows = sortedList.map((x, i) => { const medal = i === 0 ? "🥇" : i === 1 ? "🥈" : i === 2 ? "🥉" : `${i+1}.`; return ` ${medal} ${x.name} ${fmtPct(x[field])} `; }).join(""); return `
${title}
${rows}
`; } // ── Performance Snapshot Block ────────────────────────────────────────── const summary = `

🧭 Performance Snapshot — Dual KPI View

${tile("🏆 Best by Achievement", ach.best, "Achievement", "achievement", "#10b981", "#047857")} ${tile("🚀 Best by OEE", oee.best, "OEE", "oee", "#3b82f6", "#1d4ed8")} ${tile("⚠️ Lowest by Achievement", ach.worst, "Achievement", "achievement", "#f97316", "#c2410c")} ${tile("🛑 Lowest by OEE", oee.worst, "OEE", "oee", "#ef4444", "#b91c1c")}
${rankTable("📈 Achievement Ranking", ach.sorted, "achievement", "#047857")} ${rankTable("⚙️ OEE Ranking", oee.sorted, "oee", "#1d4ed8")}
📉 SBUs With Production Drop (vs previous day)
${analysis.dropped.length ? `
` + analysis.dropped.map(d => `
• ${d.name} — ${d.prevYesterday} → ${d.yesterday} (${d.drop.toFixed(2)})
` ).join("") + `
` : `
✓ No production drop detected (or first run — baseline saved).
`}
`; // ── Main email body ───────────────────────────────────────────────────── let html = `

📊 Daily SBU Production & Quality Status

Report Date: ${today}

🌐 Open Live Dashboard

${summary}`; // ── Per-SBU detailed cards (with multi-badge support) ────────────────── data.forEach(s => { const badges = []; if (ach.best && ach.best.name === s.name) badges.push(`BEST · ACH`); if (oee.best && oee.best.name === s.name) badges.push(`BEST · OEE`); if (ach.worst && ach.worst.name === s.name) badges.push(`LOW · ACH`); if (oee.worst && oee.worst.name === s.name) badges.push(`LOW · OEE`); const tag = badges.join(""); html += `

${s.name} ${tag} ${s.date}

Yesterday Prod. Achievement % OEE % NPT % MTD Prod.
${s.production.yesterday} ${s.production.achievement} ${s.production.oee} ${s.production.npt} ${s.mtdProduction}

Yesterday Downtime: ${s.npt.yesterdayDowntime}  |  MTD Downtime: ${s.npt.mtdDowntime}

${s.npt.topReasons.map(row => `` ).join("")}
Breakdown Type Reason Downtime
${row[0]||"—"} ${row[1]||"—"} ${row[2]||"—"}
Quality: 5S: ${s.quality.fiveS} | QCP %: ${s.quality.qcp} | Test Done: ${s.quality.testDone} | Audit: ${s.quality.totalAudit} (Pass: ${s.quality.pass})
Notable Activity: ${s.quality.notable || "—"}
Kaizen
✅ Complete: ${s.kaizen.complete}
🟡 Initiated: ${s.kaizen.initiated}
❌ Rejected: ${s.kaizen.rejected}
Problem Solving Cards
🔄 Ongoing: ${s.psCards.ongoing}
✅ Complete: ${s.psCards.complete}
`; }); html += `

Auto-generated from Google Sheets · View Live Dashboard

`; return html; }
